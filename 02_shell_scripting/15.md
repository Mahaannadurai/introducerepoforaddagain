## Guided Example — scripts/monitoring/health-check.sh

```bash
#!/bin/bash

# health-check.sh
# Comprehensive server health check for InfraOps
# Returns exit code 0 if healthy, 1 if warnings, 2 if critical

set -euo pipefail

source ~/devops-bash-toolkit/lib/utils.sh
source ~/devops-bash-toolkit/lib/logging.sh
source ~/devops-bash-toolkit/lib/validation.sh

# Thresholds
CPU_WARN=75
CPU_CRIT=90
MEM_WARN=80
MEM_CRIT=95
DISK_WARN=75
DISK_CRIT=90

OVERALL_STATUS=0   # 0=OK, 1=WARN, 2=CRITICAL

check_cpu() {
    local CPU_IDLE
    CPU_IDLE=$(top -bn1 | grep "Cpu(s)" | awk '{print $8}' | tr -d '%us,')
    local CPU_USED=$((100 - ${CPU_IDLE%.*}))

    if [ "$CPU_USED" -ge "$CPU_CRIT" ]; then
        print_status "FAIL" "CPU: ${CPU_USED}% (critical)"
        log_error "CPU usage critical: ${CPU_USED}%"
        [ $OVERALL_STATUS -lt 2 ] && OVERALL_STATUS=2
    elif [ "$CPU_USED" -ge "$CPU_WARN" ]; then
        print_status "WARN" "CPU: ${CPU_USED}% (elevated)"
        log_warn "CPU usage elevated: ${CPU_USED}%"
        [ $OVERALL_STATUS -lt 1 ] && OVERALL_STATUS=1
    else
        print_status "OK  " "CPU: ${CPU_USED}%"
        log_info "CPU OK: ${CPU_USED}%"
    fi
}

check_memory() {
    local MEM_TOTAL MEM_USED MEM_PCT
    MEM_TOTAL=$(free -m | awk 'NR==2 {print $2}')
    MEM_USED=$(free -m | awk 'NR==2 {print $3}')
    MEM_PCT=$(( MEM_USED * 100 / MEM_TOTAL ))

    if [ "$MEM_PCT" -ge "$MEM_CRIT" ]; then
        print_status "FAIL" "Memory: ${MEM_PCT}% (${MEM_USED}/${MEM_TOTAL} MB)"
        log_error "Memory critical: ${MEM_PCT}%"
        [ $OVERALL_STATUS -lt 2 ] && OVERALL_STATUS=2
    elif [ "$MEM_PCT" -ge "$MEM_WARN" ]; then
        print_status "WARN" "Memory: ${MEM_PCT}% (${MEM_USED}/${MEM_TOTAL} MB)"
        log_warn "Memory elevated: ${MEM_PCT}%"
        [ $OVERALL_STATUS -lt 1 ] && OVERALL_STATUS=1
    else
        print_status "OK  " "Memory: ${MEM_PCT}% (${MEM_USED}/${MEM_TOTAL} MB)"
        log_info "Memory OK: ${MEM_PCT}%"
    fi
}

check_disk() {
    for MOUNT in / /var /tmp; do
        [ -d "$MOUNT" ] || continue
        local USAGE
        USAGE=$(df "$MOUNT" | awk 'NR==2 {print $5}' | tr -d '%')

        if [ "$USAGE" -ge "$DISK_CRIT" ]; then
            print_status "FAIL" "Disk $MOUNT: ${USAGE}%"
            log_error "Disk critical on $MOUNT: ${USAGE}%"
            [ $OVERALL_STATUS -lt 2 ] && OVERALL_STATUS=2
        elif [ "$USAGE" -ge "$DISK_WARN" ]; then
            print_status "WARN" "Disk $MOUNT: ${USAGE}%"
            [ $OVERALL_STATUS -lt 1 ] && OVERALL_STATUS=1
        else
            print_status "OK  " "Disk $MOUNT: ${USAGE}%"
        fi
    done
}

# Main
print_header "Health Check — $(hostname) — $(get_timestamp)"
log_info "Health check started"

check_cpu
check_memory
check_disk

echo ""
case $OVERALL_STATUS in
    0) echo "RESULT: All systems healthy" ;;
    1) echo "RESULT: Warnings detected — review above" ;;
    2) echo "RESULT: CRITICAL — immediate attention required" ;;
esac

log_info "Health check complete. Status: $OVERALL_STATUS"
exit $OVERALL_STATUS
```

This script demonstrates everything learned so far: variables, functions, loops, conditions, command substitution, logging, error handling, exit codes, and library sourcing.
