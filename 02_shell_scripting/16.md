## Guided Example — scripts/users/user-audit.sh

```bash
#!/bin/bash

# user-audit.sh
# Lists system users, their last login, and shell
# InfraOps security audit tool

set -euo pipefail
source ~/devops-bash-toolkit/lib/utils.sh
source ~/devops-bash-toolkit/lib/logging.sh

REPORT_FILE="$HOME/devops-bash-toolkit/logs/user-audit-$(date +%Y%m%d).txt"

print_header "User Audit — $(get_timestamp)"
log_info "User audit started"

{
    printf "%-20s %-10s %-30s\n" "USERNAME" "UID" "SHELL"
    printf '%.0s-' $(seq 1 62); echo ""

    while IFS=: read -r USERNAME _ UID _ _ _ SHELL; do
        # Only show users with UID >= 1000 (real users, not system users)
        if [ "$UID" -ge 1000 ] 2>/dev/null; then
            printf "%-20s %-10s %-30s\n" "$USERNAME" "$UID" "$SHELL"
        fi
    done < /etc/passwd

} | tee "$REPORT_FILE"

TOTAL=$(awk -F: '$3 >= 1000 {count++} END {print count}' /etc/passwd)
echo ""
echo "Total regular users: $TOTAL"
echo "Report saved to: $REPORT_FILE"

log_info "User audit complete. $TOTAL users found."
```
